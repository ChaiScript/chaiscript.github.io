355  std::string load_file(const std::string &t_filename) {
356  std::ifstream infile(t_filename.c_str(), std::ios::in | std::ios::ate | std::ios::binary );
357 
358  if (!infile.is_open()) {
359  throw chaiscript::exception::file_not_found_error(t_filename);
360  }
361 
362  std::streampos size = infile.tellg();
363  infile.seekg(0, std::ios::beg);
364 
365  assert(size >= 0);
366 
367  if (size == std::streampos(0))
368  {
369  return std::string();
370  } else {
371  std::vector<char> v(static_cast<unsigned int>(size));
372  infile.read(&v[0], size);
373  return std::string(v.begin(), v.end());
374  }

655  std::string load_module(const std::string &t_module_name)
656  {
657  std::vector<exception::load_module_error> errors;
658 
659  std::vector<std::string> prefixes;
660  prefixes.push_back("lib");
661  prefixes.push_back("");
662 
663  std::vector<std::string> postfixes;
664  postfixes.push_back(".dll");
665  postfixes.push_back(".so");
666  postfixes.push_back("");
667 
668  for (size_t i = 0; i < m_modulepaths.size(); ++i)
669  {
670  for (size_t j = 0; j < prefixes.size(); ++j)
671  {
672  for (size_t k = 0; k < postfixes.size(); ++k)
673  {
674  try {
675  std::string name = m_modulepaths[i] + prefixes[j] + t_module_name + postfixes[k];
676  // std::cerr << "trying location: " << name << std::endl;
677  load_module(t_module_name, name);
678  return name;
679  } catch (const chaiscript::exception::load_module_error &e) {
680  // std::cerr << "error: " << e.what() << std::endl;
681  errors.push_back(e);
682  // Try next set
683  }
684  }
685  }
686  }
687 
688  std::string errstring;
689 
690  for (std::vector<exception::load_module_error>::const_iterator itr = errors.begin();
691  itr != errors.end();
692  ++itr)
693  {
694  if (!errstring.empty())
695  {
696  errstring += "; ";
697  }
698 
699  errstring += itr->what();
700  }

702  throw chaiscript::exception::load_module_error("Unable to find module: " + t_module_name + " Errors: " + errstring);
