157  static std::string format_types(const Const_Proxy_Function &t_func,
158  bool t_dot_notation,
159  const chaiscript::detail::Dispatch_Engine &t_ss)
160  {
161  int arity = t_func->get_arity();
162  std::vector<Type_Info> types = t_func->get_param_types();
163 
164  std::string retval;
165  if (arity == -1)
166  {
167  retval = "(...)";
168  if (t_dot_notation)
169  {
170  retval = "(Object)." + retval;
171  }
172  } else if (types.size() <= 1) {
173  retval = "()";
174  } else {
175  std::stringstream ss;
176  ss << "(";
177 
178  std::string paramstr;
179 
180  for (size_t index = 1;
181  index != types.size();
182  ++index)
183  {
184  paramstr += (types[index].is_const()?"const ":"");
185  paramstr += t_ss.get_type_name(types[index]);
186 
187  if (index == 1 && t_dot_notation)
188  {
189  paramstr += ").(";
190  if (types.size() == 2)
191  {
192  paramstr += ", ";
193  }
194  } else {
195  paramstr += ", ";
196  }
197  }
198 
199  ss << paramstr.substr(0, paramstr.size() - 2);
200 
201  ss << ")";
202  retval = ss.str();
203  }
204 
205 
206  std::shared_ptr<const dispatch::Dynamic_Proxy_Function> dynfun
207  = std::dynamic_pointer_cast<const dispatch::Dynamic_Proxy_Function>(t_func);
208 
209  if (dynfun)
210  {
211  Proxy_Function f = dynfun->get_guard();
212 
213  if (f)
214  {
215  std::shared_ptr<const dispatch::Dynamic_Proxy_Function> dynfunguard
216  = std::dynamic_pointer_cast<const dispatch::Dynamic_Proxy_Function>(f);
217  if (dynfunguard)
218  {
219  retval += " : " + format_guard(dynfunguard->get_parse_tree());
220  }
221  }
222 
223  retval += "\n Defined at " + format_location(dynfun->get_parse_tree());
224  }
225 
226  return retval;
227  }
