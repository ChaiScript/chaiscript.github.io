109  template<typename From, typename To>
110  class Dynamic_Caster
111  {
112  public:
113  static Boxed_Value cast(const Boxed_Value &t_from)
114  {
115  if (t_from.get_type_info().bare_equal(chaiscript::user_type<From>()))
116  {
117  if (t_from.is_pointer())
118  {
119  // Dynamic cast out the contained boxed value, which we know is the type we want
120  if (t_from.is_const())
121  {
122  std::shared_ptr<const To> data
123  = std::dynamic_pointer_cast<const To>(detail::Cast_Helper<std::shared_ptr<const From> >::cast(t_from, nullptr));
124  if (!data)
125  {
126  throw std::bad_cast();
127  }
128 
129  return Boxed_Value(data);
130  } else {
131  std::shared_ptr<To> data
132  = std::dynamic_pointer_cast<To>(detail::Cast_Helper<std::shared_ptr<From> >::cast(t_from, nullptr));
133 
134  if (!data)
135  {
136  throw std::bad_cast();
137  }
138 
139  return Boxed_Value(data);
140  }
141  } else {
142  // Pull the reference out of the contained boxed value, which we know is the type we want
143  if (t_from.is_const())
144  {
145  const From &d = detail::Cast_Helper<const From &>::cast(t_from, nullptr);
146  const To &data = dynamic_cast<const To &>(d);
147  return Boxed_Value(std::cref(data));
148  } else {
149  From &d = detail::Cast_Helper<From &>::cast(t_from, nullptr);
150  To &data = dynamic_cast<To &>(d);
151  return Boxed_Value(std::ref(data));
152  }
153  }
154  } else {
155  throw chaiscript::exception::bad_boxed_dynamic_cast(t_from.get_type_info(), typeid(To), "Unknown dynamic_cast_conversion");
156  }
157  }
158  };
