728  template<typename InItr>
729  Boxed_Value dispatch_with_conversions(InItr begin, const InItr &end, const std::vector<Boxed_Value> &plist,
730  const Type_Conversions &t_conversions)
731  {
732  InItr orig(begin);
733 
734  InItr matching_func(end);
735 
736  while (begin != end)
737  {
738  if (types_match_except_for_arithmetic(*begin, plist, t_conversions))
739  {
740  if (matching_func == end)
741  {
742  matching_func = begin;
743  } else {
744  // More than one function matches, not attempting
745  throw exception::dispatch_error(plist, std::vector<Const_Proxy_Function>(orig, end));
746  }
747  }
748 
749  ++begin;
750  }
751 
752  if (matching_func == end)
753  {
754  // no appropriate function to attempt arithmetic type conversion on
755  throw exception::dispatch_error(plist, std::vector<Const_Proxy_Function>(orig, end));
756  }

764  if (tis[i+1].is_arithmetic()
765  && plist[i].get_type_info().is_arithmetic()) {
766  newplist.push_back(Boxed_Number(plist[i]).get_as(tis[i+1]).bv);
767  } else {
768  newplist.push_back(plist[i]);
769  }
770  }
771 
772  try {
773  return (*(*matching_func))(newplist, t_conversions);
774  } catch (const exception::bad_boxed_cast &) {
775  //parameter failed to cast
776  } catch (const exception::arity_error &) {
777  //invalid num params
778  } catch (const exception::guard_error &) {
779  //guard failed to allow the function to execute
780  }
781

794  const std::vector<Boxed_Value> &plist, const Type_Conversions &t_conversions)
795  {
796 
797  std::multimap<size_t, const Proxy_Function_Base *> ordered_funcs;
798 
799  for (const auto &func : funcs)
800  {
801  size_t numdiffs = 0;
802  const auto arity = func->get_arity();
803 
804  if (arity == -1)
805  {
806  numdiffs = plist.size();
807  } else if (arity == static_cast<int>(plist.size())) {
808  for (size_t i = 0; i < plist.size(); ++i)
809  {
810  if (!func->get_param_types()[i+1].bare_equal(plist[i].get_type_info()))
811  {
812  ++numdiffs;
813  }
814  }
815  } else {
816  continue;
817  }
818 
819  ordered_funcs.insert(std::make_pair(numdiffs, func.get()));
820  }
821 
822  for (const auto &func : ordered_funcs )
823  {
824  try {
825  if (func.second->filter(plist, t_conversions))
826  {
827  return (*(func.second))(plist, t_conversions);
828  }
829  } catch (const exception::bad_boxed_cast &) {
830  //parameter failed to cast, try again
831  } catch (const exception::arity_error &) {
832  //invalid num params, try again
833  } catch (const exception::guard_error &) {
834  //guard failed to allow the function to execute,
835  //try again
836  }
837  }
838 
839  return detail::dispatch_with_conversions(funcs.cbegin(), funcs.cend(), plist, t_conversions);
840  }
841  }
