103  struct binary
104  {
105  template<typename T, typename U>
106  static Boxed_Value go(Operators::Opers t_oper, T &t, const U &u, const Boxed_Value &t_lhs)
107  {
108  switch (t_oper)
109  {
110  case Operators::assign:
111  t = u;
112  break;
113  case Operators::pre_increment:
114  ++t;
115  break;
116  case Operators::pre_decrement:
117  --t;
118  break;
119  case Operators::assign_product:
120  t *= u;
121  break;
122  case Operators::assign_sum:
123  t += u;
124  break;
125  case Operators::assign_quotient:
126  check_divide_by_zero(u);
127  t /= u;
128  break;
129  case Operators::assign_difference:
130  t -= u;
131  break;
132  default:
133  throw chaiscript::detail::exception::bad_any_cast();
134  }
135 
136  return t_lhs;
137  }

231  {

233  {

235  } else if (t_oper > Operators::non_const_flag && t_oper < Operators::non_const_int_flag && !t_lhs.is_const()) {
