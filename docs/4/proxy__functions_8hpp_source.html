655  template<typename InItr>
656  Boxed_Value dispatch_with_conversions(InItr begin, const InItr &end, const std::vector<Boxed_Value> &plist,
657  const Dynamic_Cast_Conversions &t_conversions)
658  {
659  InItr orig(begin);
660 
661  InItr matching_func(end);
662 
663  while (begin != end)
664  {
665  if (types_match_except_for_arithmetic(*begin, plist, t_conversions))
666  {
667  if (matching_func == end)
668  {
669  matching_func = begin;
670  } else {
671  // More than one function matches, not attempting
672  throw exception::dispatch_error(plist, std::vector<Const_Proxy_Function>(orig, end));
673  }
674  }
675 
676  ++begin;
677  }
678 
679  if (matching_func == end)
680  {
681  // no appropriate function to attempt arithmetic type conversion on
682  throw exception::dispatch_error(plist, std::vector<Const_Proxy_Function>(orig, end));
683  }
684 
685 
686  std::vector<Boxed_Value> newplist;
687  const std::vector<Type_Info> &tis = (*matching_func)->get_param_types();
688 
689  for (size_t i = 0; i < plist.size(); ++i)
690  {
691  if (tis[i+1].is_arithmetic()
692  && plist[i].get_type_info().is_arithmetic()) {
693  newplist.push_back(Boxed_Number(plist[i]).get_as(tis[i+1]).bv);
694  } else {
695  newplist.push_back(plist[i]);
696  }
697  }
698 
699  try {
700  return (*(*matching_func))(newplist, t_conversions);
701  } catch (const exception::bad_boxed_cast &) {
702  //parameter failed to cast
703  } catch (const exception::arity_error &) {
704  //invalid num params
705  } catch (const exception::guard_error &) {
706  //guard failed to allow the function to execute
707  }
708 
709  throw exception::dispatch_error(plist, std::vector<Const_Proxy_Function>(orig, end));
710 
711  }
