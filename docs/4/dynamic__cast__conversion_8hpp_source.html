76  template<typename From, typename To>
77  class Dynamic_Caster
78  {
79  public:
80  static Boxed_Value cast(const Boxed_Value &t_from)
81  {
82  if (t_from.get_type_info().bare_equal(user_type<From>()))
83  {
84  if (t_from.is_pointer())
85  {
86  // Dynamic cast out the contained boxed value, which we know is the type we want
87  if (t_from.is_const())
88  {
89  boost::shared_ptr<const To> data
90  = boost::dynamic_pointer_cast<const To>(detail::Cast_Helper<boost::shared_ptr<const From> >::cast(t_from, 0));
91  if (!data)
92  {
93  throw std::bad_cast();
94  }
95 
96  return Boxed_Value(data);
97  } else {
98  boost::shared_ptr<To> data
99  = boost::dynamic_pointer_cast<To>(detail::Cast_Helper<boost::shared_ptr<From> >::cast(t_from, 0));
100 
101  if (!data)
102  {
103  throw std::bad_cast();
104  }
105 
106  return Boxed_Value(data);
107  }
108  } else {
109  // Pull the reference out of the contained boxed value, which we know is the type we want
110  if (t_from.is_const())
111  {
112  const From &d = detail::Cast_Helper<const From &>::cast(t_from, 0);
113  const To &data = dynamic_cast<const To &>(d);
114  return Boxed_Value(boost::cref(data));
115  } else {
116  From &d = detail::Cast_Helper<From &>::cast(t_from, 0);
117  To &data = dynamic_cast<To &>(d);
118  return Boxed_Value(boost::ref(data));
119  }
120  }
121  } else {
122  throw exception::bad_boxed_dynamic_cast(t_from.get_type_info(), typeid(To), "Unknown dynamic_cast_conversion");
123  }
124  }
125  };

208  boost::shared_ptr<detail::Dynamic_Conversion> get_conversion(const Type_Info &base, const Type_Info &derived) const
209  {
210  chaiscript::detail::threading::shared_lock<chaiscript::detail::threading::shared_mutex> l(m_mutex);
211 
212  std::set<boost::shared_ptr<detail::Dynamic_Conversion> >::const_iterator itr =
213  find(base, derived);
214 
215  if (itr != m_conversions.end())
216  {
217  return *itr;
218  } else {
219  throw std::out_of_range("No such conversion exists from " + derived.bare_name() + " to " + base.bare_name());
